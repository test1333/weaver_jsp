ALTER FUNCTION [dbo].[get_some_mont_OUT](@martID int,@stockID int)
	returns decimal(18,2)
	as
	begin
		declare @res decimal(18,2)
		select  @res = ISnull(SUM(rksl),0) from uf_stock where CKID=@stockID and wlmc=@martID  and rkly=1
		
		return @res
		
	end
GO
